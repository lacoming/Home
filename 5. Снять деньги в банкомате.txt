Алгоритм: "Снять деньги в банкомате"

1) if (банкоматом никто не пользуется)
		вставить банковскую карту в банкомат;
	else{
		встать в конец очереди;
		while(твоя очередь не подошла){
			ждем;
		}
		вставить банковскую карту в банкомат;
	}
2) Нажать на экране "Снять наличные"
3)  if (банкомат выдаст сообщение "Отсутвуют наличные купюры")
		завершить алгоритм с мыслью: "Придётся топать в цивилизацию, т.к. в банкомате денег нет";
	else{
		ввести в поле сумма: "100";
		ввести пин-код банковской карты;
		if(банкомат выдаёт сообщение: "Купюры данного номинала отсутствуют, хотите продолжить операцию?"){
			нажать: "Да";
			ввести в поле сумма: "500";
			ввести пин-код банковской карты;
			if(банкомат выдаёт сообщение: "Купюры данного номинала отсутствуют, хотите продолжить операцию?"){
				нажать: "Да";
				ввести в поле сумма: "1000";
				ввести пин-код банковской карты;
				if(банкомат выдаёт сообщение: "Купюры данного номинала отсутствуют, хотите продолжить операцию?"){
					нажать: "Да";
					ввести в поле сумма: "5000";
					ввести пин-код банковской карты;
					if(банкомат выдаёт сообщение: "Купюры данного номинала отсутствуют, хотите продолжить операцию?")
						завершить алгоритм с мыслью: "Придётся топать в цивилизацию, т.к. в банкомате денег нет";
				}
				
			}
		}
		дождаться пока банкомат выдаст банковскую карту;
		забрать карту;
		дождаться пока банкомат выдаст деньги;
		забрать деньги;
	}
